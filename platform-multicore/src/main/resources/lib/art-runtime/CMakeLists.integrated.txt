# ART Runtime sources
set(art_runtime_sources
        src/actors-rts.c
        src/termination-report.c
        src/xmlParser.c
        src/xmlTrace.c
        src/device_handle.c
        )

# ART Runtime includes
set(art_runtime_header
        include/actors-config.h
        include/actors-fifo.h
        include/actors-rts.h
        include/actors_execute_network.h
        include/dll.h
        include/internal.h
        include/natives.h
        include/xmlParser.h
        include/xmlTrace.h
        include/device_handle.h
        )

# -- Set CMake module path, for finding the necessary tools
set(CMAKE_MODULE_PATH ${CMAKE_MODULE_PATH} ${CMAKE_SOURCE_DIR}/cmake)
find_package(Vivado REQUIRED)
find_package(VivadoHLS REQUIRED)
find_package(SDAccel REQUIRED)
if(SDACCEL_FOUND)
    list(APPEND extra_includes ${SDACCEL_INCLUDE_DIRS})
    list(APPEND extra_libraries ${SDACCEL_LIBRARIES})
else()
    message(FATAL_ERROR "SDAccel is not found, source SDx settings.sh")
endif()

include_directories(${extra_includes})
add_definitions(${extra_definitions})

add_library(art-runtime STATIC ${art_runtime_sources} ${art_runtime_header})

target_link_libraries(art-runtime art-native ${extra_libraries})

install(TARGETS art-runtime
        ARCHIVE DESTINATION lib)

set(extra_definitions ${extra_definitions} PARENT_SCOPE)
set(extra_includes ${extra_includes} PARENT_SCOPE)
set(extra_libraries ${extra_libraries} PARENT_SCOPE)
